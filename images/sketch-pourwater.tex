% Sketch of "pouring water" closed-loop control system
% Compile: pdflatex sketch-pourwater.tex
\documentclass[tikz,border=2mm]{standalone}
\usepackage{pgf}
\usepackage{amsmath}
\usepackage{ctex}

\setCJKmainfont{STSong}

\definecolor{ink}{RGB}{40,30,20}

\begin{document}
\begin{tikzpicture}[
    line cap=round,
    line join=round,
    ink/.style={draw=ink, very thick},
    thinink/.style={draw=ink, thick},
    dashfeedback/.style={ink, dashed, thick},
    every node/.style={font=\small}
]

% === Teapot (left) ===
% Body
\fill[ink!80!white, rounded corners=3mm] (0,3.5) ellipse (1.2 and 1.0);
\draw[ink, rounded corners=3mm] (0,3.5) ellipse (1.2 and 1.0);
% Lid
\draw[ink, rounded corners=1mm] (-0.6,4.4) -- (-0.6,4.6) -- (0.6,4.6) -- (0.6,4.4);
\draw[ink, rounded corners=1mm] (-0.45,4.6) -- (-0.2,5.0) -- (0.2,5.0) -- (0.45,4.6);
% Handle (right side of teapot)
\draw[ink, rounded corners=2mm] (0.8,3.0) .. controls (2.0,2.8) and (2.0,4.8) .. (1.0,4.0);
% Spout (left side)
\draw[ink, rounded corners=1mm] (-1.0,3.2) -- (-1.8,2.2) -- (-2.2,1.5);
\draw[ink, rounded corners=1mm] (-1.2,3.0) -- (-2.0,2.0) -- (-2.3,1.5);
% Spout opening
\draw[ink] (-2.25,1.5) -- (-2.0,1.3);

% === Water stream ===
\draw[ink, dashed] (-2.1,1.4) .. controls (-2.5,0.5) and (-2.0,-0.5) .. (-1.5,-1.5);
\draw[thinink] (-2.0,1.3) .. controls (-2.4,0.5) and (-1.9,-0.5) .. (-1.4,-1.5);
% Water droplets/splashes near cup
\foreach \x in {0.0,0.15,0.3} {
    \draw[thinink] (-1.5+\x*0.5,-1.6) .. controls (-1.4+\x*0.5,-1.8) .. (-1.5+\x*0.5,-2.0);
}

% === Cup (bottom right) ===
% Cup body
\draw[ink, rounded corners=1mm] (-2.2,-1.5) -- (-2.2,-3.0) -- (-0.2,-3.0) -- (-0.2,-1.5);
% Water surface
\draw[ink] (-2.15,-2.0) -- (-0.25,-2.0);
% Water fill (hatching style)
\foreach \y in {-2.2,-2.4,-2.6,-2.8} {
    \draw[thinink] (-2.1,\y) -- (-0.3,\y);
}
% Cup handle (right)
\draw[ink, rounded corners=1mm] (-0.2,-2.2) .. controls (0.4,-2.0) and (0.4,-2.8) .. (-0.2,-2.6);
% Cup rim
\draw[ink, rounded corners=1mm] (-2.3,-1.5) -- (-2.3,-1.4) -- (-0.1,-1.4) -- (-0.1,-1.5);
% Base
\draw[ink, rounded corners=1mm] (-2.0,-3.0) -- (-2.0,-3.1) -- (-0.4,-3.1) -- (-0.4,-3.0);

% === Table surface ===
\draw[ink] (-3.5,-3.1) -- (2.0,-3.1);
\draw[thinink] (-3.5,-3.15) -- (2.0,-3.15);

% === Arm (from teapot handle to person) ===
% Simplified arm/shoulder on the right
\draw[ink, rounded corners=3mm] (1.5,3.8) .. controls (2.5,3.5) and (3.0,3.0) .. (3.5,2.5);
\draw[ink, rounded corners=2mm] (1.5,4.0) .. controls (2.2,4.2) and (2.8,3.8) .. (3.5,2.5);
% Sleeve cuff
\draw[ink, rounded corners=1mm] (2.0,3.9) -- (2.0,3.6) -- (2.5,3.5) -- (2.5,3.8);

% === Simplified person/head (right side) ===
% Head
\draw[ink, rounded corners=2mm] (3.2,1.5) circle (0.7);
% Body
\draw[ink, rounded corners=2mm] (3.2,0.8) -- (3.2,-0.5) -- (3.8,-0.5) -- (3.8,0.8) -- cycle;
\draw[ink, rounded corners=2mm] (2.6,-0.5) -- (3.2,-0.5) -- (3.2,0.8);
% Eye (looking at the cup)
\draw[ink] (3.5,1.7) -- (3.7,1.6);
\draw[thinink] (3.7,1.6) circle (0.12);
% Eyebrow (slight tilt - concentrating)
\draw[ink] (3.5,1.9) -- (3.7,1.85);
% Nose
\draw[thinink] (3.8,1.5) -- (3.95,1.3) -- (3.8,1.2);

% === Feedback arrow (eye → brain → arm) ===
% Visual feedback: from cup to eye (observation)
\draw[dashfeedback,->] (-0.5,-1.0) .. controls (1.0,-0.5) and (2.0,0.5) .. (3.3,1.2);
\node[ink] at (1.8, -0.3) {\scriptsize 反馈(视觉)};

% === Labels ===
\node[ink, below] at (-1.2,-3.6) {水杯(被控对象)};
\node[ink, above] at (0,5.2) {茶壶(执行器)};
\node[ink, right] at (4.5,1.0) {大脑(控制器)};
\node[ink, right] at (4.5,-0.8) {手臂};

% === Annotation box (top right) ===
\draw[ink, rounded corners=1mm] (4.8,3.5) rectangle (8.0,5.5);
\node[ink, align=left, anchor=north west] at (5.0,5.3) {
    \textbf{闭环控制要素}\\
    \small 参考输入：期望水位\\
    \small 控制器：大脑\\
    \small 执行器：手臂/茶壶\\
    \small 被控对象：水杯水位\\
    \small 传感器：眼睛(视觉)\\
    \small 反馈：视觉信号
};

\end{tikzpicture}
\end{document}
